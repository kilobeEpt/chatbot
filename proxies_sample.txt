# Sample Proxy Configuration File
# Supported formats:
# 1. scheme://user:pass@host:port
# 2. host:port
# 3. host:port:user:pass

# Anonymous proxies
proxy1.example.com:8080
proxy2.example.com:3128
192.168.1.100:1080

# Authenticated proxies
username:password@secure-proxy.com:8080
user123:pass456@10.0.0.50:3128

# With explicit scheme
http://user:pass@auth-proxy.example.com:8080
https://encrypted-proxy.example.com:3128

# Mixed examples
proxy-server-1.com:1080
user:pass@proxy-server-2.com:8080
https://user:password@secure-proxy-3.com:3128
192.168.100.10:1080
192.168.100.11:1080:user:pass
